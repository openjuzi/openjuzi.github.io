      public class HWID extends JFrame{
    Scanner sc = new Scanner(System.in);

    private PrintWriter writer ;
    Socket socket;
    private JTextArea area = new JTextArea();
    private  JTextField test = new JTextField();


    public HWID(){
        setTitle("HWID验证");
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        Container c = getContentPane();
        JScrollPane scrollPane = new JScrollPane(area);
        getContentPane().add(scrollPane,BorderLayout.CENTER);
        c.add(test,"South");

        test.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                writer.println(test.getText().trim());
                area.append(test.getText()+'\n');
                test.setText("");


            }
        });


    }

    public void connect(){
        area.append("尝试连接到验证服务器\n");
        try {
            while (true) {
                socket = new Socket("127.0.0.1", 25565); //ip端口在这修改哦！
                writer = new PrintWriter(socket.getOutputStream(), true);
                area.append("验证服务器在线！");
                area.append("请输入你的账号 授权码(联系dev) 格式: xxx xxx");
            }

            }catch(IOException exception){

                exception.printStackTrace();
                area.append("验证服务器不在线！\n请联系DEV进行解决！");
                area.append("重启客户端进行尝试");


        }
    }

    public static void main(String[] args) {

        HWID client = new HWID();
        client.setSize(200,150);
        client.setVisible(true);
        client.connect();

    }
}

      public class ServerInfo {
    Boolean stop = false;
    Scanner sc = new Scanner(System.in);
    private ServerSocket server;
    private Socket socket;
    Boolean run = null; //拒绝或者接受 刚开始赋值为null
    void start() {
        System.out.println("输入你要监听的端口号");
        int port = sc.nextInt();
        int sqm = 1;
        try{

            server = new ServerSocket(port);
            while (true){
                System.out.println("正在接受客户端的HWID....");
                socket = server.accept();
                BufferedReader reader = new BufferedReader(new InputStreamReader(socket.getInputStream()));

                    String HWID = reader.readLine();
                    if ("exit".equals(HWID)){
                        System.out.println("客户端退出！");
                        break;
                    }
                    System.out.println("用户账号: "+HWID);


            }
            server.close();
            socket.close();

        }catch (IOException e){
            System.out.println("发生了未知的错误！");
            if (stop == true){
                System.out.println("程序退出");
            }
        }
    }

    public static void main(String[] args) {

        InetAddress ip ;
        try{
            ip = InetAddress.getLocalHost();
            String Localname = ip.getHostName();
            String Localip = ip.getHostAddress();
            System.out.println("服务机: "+ Localname + " 您的ip是: " + Localip);
            ServerInfo tcp = new ServerInfo();
            tcp.start();

        }catch (UnknownHostException e){
            System.out.println("发生了未知的错误");
        }
    }
}
